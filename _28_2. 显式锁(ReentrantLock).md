# 显式锁(ReentrantLock)
---
```java
import java.util.concurrent.locks.ReentrantLock;

class Main {
  static Object o = new Object();

  public static void main(String[] args) {

    Runnable run = new MyThread1();
    Thread a = new Thread(run);
    a.start();
    Thread b = new Thread(run);
    b.start();
    Thread c = new Thread(run);
    c.start();

    //a.stop();
    //b.stop();
    //c.stop();

    // System.out.print("\n"+Thread.currentThread().getName()+"\n");

  }
}

class MyThread1 implements Runnable {
  private int breakfast = 10;
  String name = null;
  static Object o = new Object();

  // 显式锁 l initiate
  ReentrantLock l = new ReentrantLock();

  public MyThread1(String name) {
    this.name = name;
  }

  public MyThread1() {
  }

  @Override
  public void run() {

    while (breakfast > 0) {
      //lock before dealing with data
      l.lock();
      if (breakfast > 0) {
        System.out.print("Selling Breakfast..." + "\n");
        try {
          Thread.sleep(1000);
        } catch (InterruptedException e) {
          e.printStackTrace();
        }
        breakfast--;
        System.out.print("Current breakfast number: " + this.breakfast + "\n");
      } else {
        break;
      }
      //unlock after dealing with data
      l.unlock();
    }
    
    return;
  }

}
```
