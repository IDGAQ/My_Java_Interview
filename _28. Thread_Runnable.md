```java
class MyThread1 implements Runnable {
  private int breakfast = 10;
  private String name;

  public MyThread1(String name) {
    this.name = name;
  }

  public MyThread1() {
  }

  public void run() {
    for (int i = 0; i < 500; i++) {
      if (this.breakfast > 0) {
        System.out.println(this.name + "卖早餐---->" + (this.breakfast--));
      }
    }
  }
}
```
